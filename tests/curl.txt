# User tokens refresh url
curl \
    -X 'POST' http://127.0.0.1:8000/api/token/refresh/ \
    -H 'accept: application/json' \
    -H 'Content-Type: application/json' \
    -d '{"refresh": "token" }'\ 
    | json


# User JWT Login/Auth url
curl \
    -X POST http://127.0.0.1:8000/api/v1/auth/jwt/
    -H 'accept: application/json' \
    -H 'Content-type: application/json' \
    -d '{"nickname": "test_boy", "password": "required" }' \
    | json



# Anime GET List. Access token needed
curl \
    -X 'GET' http://127.0.0.1:8000/api/v1/animes/ \
    -H "Accept: application/json" \
    -H "Authorization: Bearer {acces_token}" \
    | json
